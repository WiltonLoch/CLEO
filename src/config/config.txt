### Initialisation Files parameters ###
initSDs_filename = ./share/dimlessSDsinit.dat       # binary filename for initialisation of SDs
grid_filename = ./share/dimlessGBxboundaries.dat    # binary filename for GBx boundaries

### Output Data parameters ###
setuptxt = ./bin/solsetup.txt               # name of .txt output file to copy setup to
zarrbasedir = ./bin/solSDMdata.zarr         # zarr store base directory
maxchunk = 1250000                          # maximum no. of elements in chunks of zarr store array

### SDM parameters ###
# timestepping parameters #
cond_maxiters = 100                     # maximum no. iterations of Newton Raphson Method
cond_rtol = 1e-1                        # relative tolerance for implicit euler integration
cond_atol = 100                         # abolute tolerance for implicit euler integration
CONDTSTEP = 1                           # time between SD condensation events [s]
COLLTSTEP = 1                           # time between SD collision events [s]
MOTIONTSTEP = 5                         # time between SDM motion [s]
COUPLTSTEP = 30                         # time between SDM observations and between thermodynamic couplings [s]
T_END = 240                             # time span of integration [s]

# SDs parameters #
nSDsvec = 1024                          # initial no. elements in SDs' vector (=total initial no. of SDs)
SDnspace = 1                            # no. of spatial coordinates of SDs (=dimension of model)
wetradiiinit = false                    # set initial SD radii to equilibrium wet radius
doAlterThermo = false                   # enable condensation to alter the thermodynamic state

### Read in Thermodynamics File parameters ###
press_filename = ./share/dimlessthermodynamics_press.dat          # binary filename for pressure
temp_filename = ./share/dimlessthermodynamics_temp.dat            # binary filename for temperature
qvap_filename = ./share/dimlessthermodynamics_qvap.dat            # binary filename for vapour mixing ratio
qcond_filename = ./share/dimlessthermodynamics_qcond.dat          # binary filename for liquid mixing ratio
wvel_filename = ./share/dimlessthermodynamics_wvel.dat            # binary filename for vertical (z) velocity
uvel_filename = ./share/dimlessthermodynamics_uvel.dat            # binary filename for horizontal x velocity
vvel_filename = ./share/dimlessthermodynamics_vvel.dat            # binary filename for horizontal y velocity
